476 /*==============================================================================================================================
477 
478  The nDoSimulation member function of CSimulation sets up and runs the simulation
479 
480 ==============================================================================================================================*/
481 int CSimulation::nDoSimulation(int nArg, char* pcArgv[])
482 {
483 #ifdef RANDCHECK
484  CheckRand();
485  return RTN_OK;
486 #endif
487 
488  // ================================================== initialization section ================================================
489  // Hello, World!
490  AnnounceStart();
491 
492  // Start the clock ticking
493  StartClock();
494 
495  // Find out the folder in which the CoastalME executable sits, in order to open the .ini file (they are assumed to be in the same folder)
496  if (! bFindExeDir(pcArgv[0]))
497  return (RTN_ERR_CMEDIR);
498 
499  // Deal with command-line parameters
500  int nRet = nHandleCommandLineParams(nArg, pcArgv);
501  if (nRet != RTN_OK)
502  return (nRet);
503 
504  // OK, we are off, tell the user about the licence and the start time
505  AnnounceLicence();
506 
507  // Read the .ini file and get the name of the run-data file, and path for output etc.
508  if (! bReadIniFile())
509  return (RTN_ERR_INI);
510 
511  // We have the name of the run-data input file, so read it
512  if (! bReadRunDataFile())
513  return RTN_ERR_RUNDATA;
514 
515  // Check raster GIS output format
516  if (! bCheckRasterGISOutputFormat())
517  return (RTN_ERR_RASTER_GIS_OUT_FORMAT);
518 
519  // Check vector GIS output format
520  if (! bCheckVectorGISOutputFormat())
521  return (RTN_ERR_VECTOR_GIS_OUT_FORMAT);
522 
523  // Open log file
524  if (! bOpenLogFile())
525  return (RTN_ERR_LOGFILE);
526 
527  // Set up the time series output files
528  if (! bSetUpTSFiles())
529  return (RTN_ERR_TSFILE);
530 
531  // Initialize the random number generators
532  InitRand0(m_ulRandSeed[0]);
533  InitRand1(m_ulRandSeed[1]);
534 
535  // If we are doing Savitzky-Golay smoothing of the vector coastline(s), calculate the filter coefficients
536  if (m_nCoastSmooth == SMOOTH_SAVITZKY_GOLAY)
537  CalcSavitzkyGolayCoeffs();
538 
539  // Create the raster grid object
540  m_pRasterGrid = new CGeomRasterGrid(this);
541 
542  // Read in the basement layer (NOTE MUST HAVE THIS FILE), create the raster grid, then read in the basement DEM data to the array
543  AnnounceReadBasementDEM();
544  nRet = nReadBasementDEMData();
545  if (nRet != RTN_OK)
546  return nRet;
547 
548  m_ulNumCells = m_nXGridMax * m_nYGridMax;
549 
550  // Mark edge cells, as defined by the basement layer
551  nRet = nMarkBoundingBoxEdgeCells();
552  if (nRet != RTN_OK)
553  return nRet;
554 
555 // // DEBUG CODE =================================================
556 // for (int n = 0; n < m_VEdgeCell.size(); n++)
557 // {
558 // LogStream << "[" << m_VEdgeCell[n].nGetX() << "][" << m_VEdgeCell[n].nGetY() << "] = {" << dGridCentroidXToExtCRSX(m_VEdgeCell[n].nGetX()) << ", " << dGridCentroidYToExtCRSY(m_VEdgeCell[n].nGetY()) << "} " << m_VEdgeCellEdge[n] << endl;
559 // }
560 // // DEBUG CODE =================================================
561 
562  // If we are using the default cell spacing, then now that we know the size of the raster cells, we can set the size of profile spacing in m
563  if (m_dCoastNormalAvgSpacing == 0)
564  m_dCoastNormalAvgSpacing = MIN_PROFILE_SPACING * m_dCellSide;
565  else
566  {
567  // The user specified a profile spacing, is this too small?
568  m_nCoastNormalAvgSpacing = nRound(m_dCoastNormalAvgSpacing / m_dCellSide);
569 
570  if (m_nCoastNormalAvgSpacing < MIN_PROFILE_SPACING)
571  {
572  cerr << ERR << "profile spacing was specified as " << m_dCoastNormalAvgSpacing << " m, which is " << m_nCoastNormalAvgSpacing << " cells. Polygon creation works poorly if profile spacing is less than " << MIN_PROFILE_SPACING << " cells, i.e. " << MIN_PROFILE_SPACING * m_dCellSide << " m" << endl;
573 
574  LogStream << ERR << "profile spacing was specified as " << m_dCoastNormalAvgSpacing << " m, which is " << m_nCoastNormalAvgSpacing << " cells. Polygon creation works poorly if profile spacing is less than " << MIN_PROFILE_SPACING << " cells, i.e. " << MIN_PROFILE_SPACING * m_dCellSide << " m" << endl;
575 
576  return RTN_ERR_PROFILESPACING;
577  }
578  }
579 
580  // We have at least one filename for the first layer, so add the correct number of layers. Note the the number of layers does not change during the simulation: however layers can decrease in thickness until they have zero thickness
581  AnnounceAddLayers();
582  for (int nX = 0; nX < m_nXGridMax; nX++)
583  for (int nY = 0; nY < m_nYGridMax; nY++)
584  m_pRasterGrid->m_Cell[nX][nY].AppendLayers(m_nLayers);
585 
586  // Tell the user what is happening then read in the layer files
587  AnnounceReadRasterFiles();
588  for (int nLayer = 0; nLayer < m_nLayers; nLayer++)
589  {
590  // Read in the initial fine unconsolidated sediment depth file(s)
591  AnnounceReadInitialFineUnconsSedGIS(nLayer);
592  nRet = nReadRasterGISData(FINE_UNCONS_RASTER, nLayer);
593  if (nRet != RTN_OK)
594  return (nRet);
595 
596  // Read in the initial sand unconsolidated sediment depth file
597  AnnounceReadInitialSandUnconsSedGIS(nLayer);
598  nRet = nReadRasterGISData(SAND_UNCONS_RASTER, nLayer);
599  if (nRet != RTN_OK)
600  return (nRet);
601 
602  // Read in the initial coarse unconsolidated sediment depth file
603  AnnounceReadInitialCoarseUnconsSedGIS(nLayer);
604  nRet = nReadRasterGISData(COARSE_UNCONS_RASTER, nLayer);
605  if (nRet != RTN_OK)
606  return (nRet);
607 
608  // Read in the initial fine consolidated sediment depth file
609  AnnounceReadInitialFineConsSedGIS(nLayer);
610  nRet = nReadRasterGISData(FINE_CONS_RASTER, nLayer);
611  if (nRet != RTN_OK)
612  return (nRet);
613 
614  // Read in the initial sand consolidated sediment depth file
615  AnnounceReadInitialSandConsSedGIS(nLayer);
616  nRet = nReadRasterGISData(SAND_CONS_RASTER, nLayer);
617  if (nRet != RTN_OK)
618  return (nRet);
619 
620  // Read in the initial coarse consolidated sediment depth file
621  AnnounceReadInitialCoarseConsSedGIS(nLayer);
622  nRet = nReadRasterGISData(COARSE_CONS_RASTER, nLayer);
623  if (nRet != RTN_OK)
624  return (nRet);
625  }
626 
627  // Read in the initial suspended sediment depth file
628  AnnounceReadInitialSuspSedGIS();
629  nRet = nReadRasterGISData(SUSP_SED_RASTER, 0);
630  if (nRet != RTN_OK)
631  return (nRet);
