44 /*
45  Notes re. co-ordinate systems used
46 
47  1. In the raster CRS, cell[0][0] is at the top left (NW) corner of the grid. Raster grid co-oordinate [0][0] is actually the top left (NW) corner of this cell.
48 
49  2. We assume that the grid CRS and external CRS have parallel axes. If they have not, see http://www.gdal.org/classGDALDataset.html which says that:
50 
51  To convert between pixel/line (P,L) raster space, and projection coordinates (Xp,Yp) space
52  Xp = padfTransform[0] + P*padfTransform[1] + L*padfTransform[2];
53  Yp = padfTransform[3] + P*padfTransform[4] + L*padfTransform[5];
54 
55  In a north-up image, padfTransform[1] is the pixel width, and padfTransform[5] is the pixel height. The upper left corner of the upper left pixel is at position
56  (padfTransform[0], padfTransform[3]).
57 
58  3. Usually, raster grid CRS values are integer, i.e. they refer to a point which is at the centroid of a cell. They may also be -ve or greater than m_nXGridMax-1 i.e. may refer to a point which lies outside any cell of the raster grid.
59 
60 */

257 /*==============================================================================================================================
258 
259  Given two points in the grid CRS (the points assumed not to be coincident), this routine modifies the value of the second point so that it is on a line joining the original two points and is a valid cell within the raster grid
260 
261  However in some cases (e.g. if the first point is at the edge of the valid part of the raster grid) then the second cell will be coincident with the first cell, and the line joining them is thus of zero length. The calling routine has to be able to handle this
262 
263 ===============================================================================================================================*/
264 void CSimulation::KeepWithinValidGrid(int nX0, int nY0, int& nX1, int& nY1) const
265 {
266  // Safety check: make sure that the first point is within the valid grid
267  if (nX0 >= m_nXGridMax)
268  nX0 = m_nXGridMax-1;
269  else if (nX0 < 0)
270  nX0 = 0;
271 
272  if (nY0 >= m_nYGridMax)
273  nY0 = m_nYGridMax-1;
274  else if (nY0 < 0)
275  nY0 = 0;
276 
277  // OK let's go
278  int
279  nDiffX = nX0 - nX1,
280  nDiffY = nY0 - nY1;
281 
282  if (nDiffX == 0)
283  {
284  // The two points have the same x co-ords, so we just need to constrain the y co-ord
285  if (nY1 < nY0)
286  {
287  nY1 = -1;
288 
289  do
290  {
291  nY1++;
292  }
293  while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
294 
295  return;
296  }
297  else
298  {
299  nY1 = m_nYGridMax;
300 
301  do
302  {
303  nY1--;
304  }
305  while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
306 
307  return;
308  }
309  }
310  else if (nDiffY == 0)
311  {
312  // The two points have the same y co-ords, so we just need to constrain the x co-ord
313  if (nX1 < nX0)
314  {
315  nX1 = -1;
316 
317  do
318  {
319  nX1++;
320  }
321  while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
322 
323  return;
324  }
325  else
326  {
327  nX1 = m_nXGridMax;
328 
329  do
330  {
331  nX1--;
332  }
333  while (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue());
334 
335  return;
336  }
337  }
338  else
339  {
340  // The two points have different x co-ords and different y co-ords, so we have to work harder. First find which of the co-ords is the greatest distance outside the grid, and constrain that co-ord for efficiency (since this will reduce the number of times round the loop). Note that both may be inside the grid, if the incorrect co-ord is in the invalid margin, in which case arbitrarily contrain the x co-ord
341  int
342  nXDistanceOutside = 0,
343  nYDistanceOutside = 0;
344 
345  if (nX1 < 0)
346  nXDistanceOutside = -nX1;
347  else if (nX1 >= m_nXGridMax)
348  nXDistanceOutside = nX1 - m_nXGridMax + 1;
349 
350  if (nY1 < 0)
351  nYDistanceOutside = -nY1;
352  else if (nY1 >= m_nYGridMax)
353  nXDistanceOutside = nY1 - m_nYGridMax + 1;
354 
355  if (nXDistanceOutside >= nYDistanceOutside)
356  {
357  // Constrain the x co-ord
358  if (nX1 < nX0)
359  {
360  // The incorrect x co-ord is less than the correct x co-ord: constrain it and find the y co-ord
361  nX1 = -1;
362 
363  do
364  {
365  nX1++;
366 
367  nY1 = nY0 + nRound(((nX1 - nX0) * nDiffY) / static_cast<double>(nDiffX));
368  }
369  while ((nY1 < 0) || (nY1 >= m_nYGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
370 
371  return;
372  }
373 
374  else
375  {
376  // The incorrect x co-ord is greater than the correct x-co-ord: constrain it and find the y co-ord
377  nX1 = m_nXGridMax;
378 
379  do
380  {
381  nX1--;
382 
383  nY1 = nY0 + nRound(((nX1 - nX0) * nDiffY) / static_cast<double>(nDiffX));
384  }
385  while ((nY1 < 0) || (nY1 >= m_nYGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
386 
387  return;
388  }
389  }
390  else
391  {
392  // Constrain the y co-ord
393  if (nY1 < nY0)
394  {
395  // The incorrect y co-ord is less than the correct y-co-ord: constrain it and find the x co-ord
396  nY1 = -1;
397 
398  do
399  {
400  nY1++;
401 
402  nX1 = nX0 + nRound(((nY1 - nY0) * nDiffX) / static_cast<double>(nDiffY));
403  }
404  while ((nX1 < 0) || (nX1 >= m_nXGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
405 
406  return;
407  }
408 
409  else
410  {
411  // The incorrect y co-ord is greater than the correct y co-ord: constrain it and find the x co-ord
412  nY1 = m_nYGridMax;
413 
414  do
415  {
416  nY1--;
417 
418  nX1 = nX0 + nRound(((nY1 - nY0) * nDiffX) / static_cast<double>(nDiffY));
419  }
420  while ((nX1 < 0) || (nX1 >= m_nXGridMax) || (m_pRasterGrid->m_Cell[nX1][nY1].bBasementElevIsMissingValue()));
421 
422  return;
423  }
424  }
425  }
426 }

572 {

574  int
575  nSize = static_cast<int>(pVIn->size()),
576  nX0 = 0, // Current vertex X
577  nY0 = 0, // Current vertex Y
578  nX1 = 0, // Next vertex X
579  nY1 = 0; // Next vertex Y
580 
581  double
582  dA = 0, // Partial signed area
583  dSignedArea = 0.0;
584 
585  // For all vertices except last
586  for (int i = 0; i < nSize-1; ++i)
587  {
588  nX0 = pVIn->at(i).nGetX();
589  nY0 = pVIn->at(i).nGetY();
590  nX1 = pVIn->at(i+1).nGetX();
591  nY1 = pVIn->at(i+1).nGetY();
592 
593  dA = (nX0 * nY1) - (nX1 * nY0);
594  dSignedArea += dA;
595  PtiCentroid.AddXAddY((nX0 + nX1) * dA, (nY0 + nY1) * dA);
596  }
597 
598  // Do last vertex separately to avoid performing an expensive modulus operation in each iteration
599  nX0 = pVIn->at(nSize-1).nGetX();
600  nY0 = pVIn->at(nSize-1).nGetY();
601  nX1 = pVIn->at(0).nGetX();
602  nY1 = pVIn->at(0).nGetY();
603 
604  dA = (nX0 * nY1) - (nX1 * nY0);
605  dSignedArea += dA;
606  PtiCentroid.AddXAddY((nX0 + nX1) * dA, (nY0 + nY1) * dA);
607 
608  dSignedArea *= 0.5;
609  PtiCentroid.DivXDivY(6.0 * dSignedArea, 6.0 * dSignedArea);
610 
611  return PtiCentroid;
612 }

764 /*==============================================================================================================================

766  Returns the signed angle BAC (in radians) subtended between three CGeom2DIPoints B A C. From http://stackoverflow.com/questions/3057448/angle-between-3-vertices
767 
768 *==============================================================================================================================*/
769 double CSimulation::dAngleSubtended(CGeom2DIPoint const* pPtiA, CGeom2DIPoint const* pPtiB, CGeom2DIPoint const* pPtiC)
770 {
771  double
772  dXDistBtoA = pPtiB->nGetX() - pPtiA->nGetX(),
773  dYDistBtoA = pPtiB->nGetY() - pPtiA->nGetY(),
774  dXDistCtoA = pPtiC->nGetX() - pPtiA->nGetX(),
775  dYDistCtoA = pPtiC->nGetY() - pPtiA->nGetY(),
776  dDotProduct = dXDistBtoA * dXDistCtoA + dYDistBtoA * dYDistCtoA,
777  dPseudoCrossProduct = dXDistBtoA * dYDistCtoA - dYDistBtoA * dXDistCtoA,
778  dAngle = atan2(dPseudoCrossProduct, dDotProduct);
779 
780  return dAngle;
